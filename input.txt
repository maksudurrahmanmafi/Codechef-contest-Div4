#include <bits/stdc++.h>
using namespace std;

void solve() {
    int n;
    cin >> n;

    // N < 3 হলে ৩টি পথ পাওয়া অসম্ভব
    if (n < 3) {
        cout << -1 << "\n";
        return;
    }

    // n x n গ্রিড তৈরি করে সব 0 দিয়ে পূর্ণ করা
    vector<vector<int>> grid(n, vector<int>(n, 0));

    // প্রথম সারির সব সেল 1 করা
    for (int j = 0; j < n; j++) {
        grid[0][j] = 1;
    }

    // শেষ দুটি কলামের বাকি সেলগুলো 1 করা
    for (int i = 1; i < n; i++) {
        grid[i][n - 2] = 1;
        grid[i][n - 1] = 1;
    }

    // গ্রিড প্রিন্ট করা
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << grid[i][j] << (j == n - 1 ? "" : " ");
        }
        cout << "\n";
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
