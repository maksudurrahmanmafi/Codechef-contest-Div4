#include <bits/stdc++.h>
using namespace std;

bool isGood(int a, int b) {
    return (a ^ b) == abs(a - b);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int T;
    cin >> T;
    while (T--) {
        int N;
        cin >> N;
        vector<int> A(N);
        for (int i = 0; i < N; i++) cin >> A[i];

        vector<int> subseq;
        sort(A.begin(), A.end());

        for (int i = 0; i < N; i++) {
            bool canAdd = true;
            for (int x : subseq) {
                if (!isGood(x, A[i])) {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd) subseq.push_back(A[i]);
        }

        cout << subseq.size() << "\n";
    }

    return 0;
}
