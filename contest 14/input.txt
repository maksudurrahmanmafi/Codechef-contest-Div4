#include <iostream>
#include <vector>
#include <set>

using namespace std;

void solve() {
    int n;
    cin >> n;
    
    vector<int> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    
    set<int> seen;
    int pairs = 0;
    
    for (int i = 0; i < n; i++) {
        // যদি সংখ্যাটি আগে দেখা হয়ে থাকে, তবে একটি জোড়া তৈরি হলো
        if (seen.count(a[i])) {
            pairs++;
            seen.clear(); // নতুন জোড়া খোঁজার জন্য মেমোরি ক্লিয়ার
        } else {
            // অন্যথায় সংখ্যাটি মনে রাখি
            seen.insert(a[i]);
        }
    }
    
    // প্রতিটি জোড়ায় ২টি করে উপাদান থাকে
    cout << pairs * 2 << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
