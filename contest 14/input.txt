#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    cin>>t;
    while(t--){
    int N;
     
        cin >> N;
    
        map<int, int> freq;
    
        for (int i = 0; i < N; i++) {
            int x;
            cin >> x;
            freq[x]++;
        }
    
        int bestColor = -1;
        int maxFreq = 0;
    
        for (auto it : freq) {
            int color = it.first;
            int count = it.second;
    
            if (count > maxFreq) {
                maxFreq = count;
                bestColor = color;
            }
        }
    
        cout << bestColor << "\n";
    }
    return 0;
}
