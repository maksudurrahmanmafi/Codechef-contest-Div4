#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

void solve() {
    int n;
    cin >> n;
    vector<int> a(n);
    
    int even_count = 0;
    int odd_count = 0;

    for (int i = 0; i < n; i++) {
        cin >> a[i];
        // পজিটিভ বা নেগেটিভ নির্বিশেষে জোড়-বিজোড় চেক
        if (abs(a[i]) % 2 == 0) {
            even_count++;
        } else {
            odd_count++;
        }
    }

    // যদি সবগুলোই জোড় হয় অথবা সবগুলোই বিজোড় হয়
    if (even_count == n || odd_count == n) {
        cout << "YES" << endl;
    } else {
        cout << "NO" << endl;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
