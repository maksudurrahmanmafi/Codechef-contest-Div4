#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void solve() {
    int n;
    long long k;
    cin >> n >> k;
    vector<long long> p(n);
    for (int i = 0; i < n; i++) cin >> p[i];

    vector<int> dp(n + 1, 0);
    
    for (int i = 1; i <= n; i++) {
        dp[i] = dp[i - 1]; // স্কিপ করার অপশন
        for (int j = 1; j < i; j++) {
            if (abs(p[i - 1] - p[j - 1]) > k) {
                dp[i] = max(dp[i], 1 + dp[j - 1]);
            }
        }
    }
    cout << dp[n] << endl;
}
int main(){
    int t;
    cin>>t;
    while(t--){
     solve();
    }
}
